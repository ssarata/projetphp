<?php
	namespace Src\Models;
	class Model
	{
		
		protected static function load_env(string $path)
		{
		
		$lines = file($path, FILE_IGNORE_NEW_LINES); // Lecture du fichier .env ligne par ligne dans un tableau
		
		  
		
		foreach ($lines as $line) { // Parcours de chaque ligne du fichier .env
		
		list($name, $value) = explode("=", $line, 2); // Sparation de chaque ligne en cl et valeur
		
		$_ENV[$name] = $value; // Attribution de la valeur  la cl dans le tableau $_ENV
		
		}
		
		return $_ENV; // Retourne le tableau $_ENV contenant les variables d'environnement
		
		}
		
		public function __construct()
		
		{ /** */
		
		}
		
		protected static function getDb()
		
		{
		
		$_ENV = Model::load_env(__DIR__ . "/../../.env"); // Chargement des variables d'environnement
		
		try {
		
		// Tentative de connexion  la base de donnes
		
		$bdd = new \PDO('pgsql:host=' . $_ENV['DB_HOST'] . ';dbname=' . $_ENV['DB_NAME'] . ';port=' . $_ENV['DB_PORT'], $_ENV['DB_USER'], $_ENV['DB_PWD']);
		
		} catch (\Exception $e) {
		
		// En cas d'chec de connexion, affiche l'erreur
		
		die('Erreur : ' . $e->getMessage());
		
		}
		
		return $bdd; // Retourne l'objet PDO reprsentant la connexion  la base de donnes
		
		}
		
		  
		
		protected static function dbCreate(string $table, array $data)
		
		{
		
		$bdd = self::getDb(); // Rcupre la connexion  la base de donnes
		
		// Construction de la requte d'insertion
		
		$req = "INSERT INTO " . $table;
		
		$req .= " (" . implode(", ", array_keys($data)) . ")";
		
		$req .= " VALUES (:" . implode(", :", array_keys($data)) . ") ";
		
		$response = $bdd->prepare($req); // Prpare la requte
		
		$response->execute($data); // Excute la requte en liant les valeurs
		
		return $bdd->lastInsertId(); // Retourne l'identifiant de la dernire insertion
		
		}
		
		  
		
		protected static function dbDelete(string $table, array $data)
		
		{
		
		$bdd = self::getDb(); // Rcupre la connexion  la base de donnes
		
		// Construction de la requte de suppression
		
		$req = "DELETE FROM " . $table . " WHERE " . array_keys($data)[0] . " = :" . array_keys($data)[0];
		
		$response = $bdd->prepare($req); // Prpare la requte
		
		$response->execute($data); // Excute la requte en liant les valeurs
		
		return; // Retourne rien
		
		}
		
		  
		
		protected static function dbFind(string $table, array $request = null)
		
		{
		
		$bdd = self::getDb(); // Rcupre la connexion  la base de donnes
		
		$req = "SELECT * FROM " . $table; // Construction de la requte de slection
		
		if (isset($request)) { // Si des conditions de recherche sont spcifies
		
		$req .= " WHERE "; // Ajoute la clause WHERE  la requte
		
		$reqOrder = ''; // Initialise la clause ORDER BY
		
		foreach ($request as $r) { // Parcours des conditions de recherche
		
		switch ($r[0]): // Analyse de la premire partie de la condition
		
		case "orderBy": // Si c'est une clause ORDER BY
		
		$reqOrder = " ORDER BY " . htmlspecialchars($r[1]) . "" . htmlspecialchars($r[2]); // Construction de la clause ORDER BY
		
		break;
		
		default: // Pour d'autres conditions
		
		$req .= htmlspecialchars($r[0]) . htmlspecialchars($r[1]) . htmlspecialchars($r[2]); // Ajoute la condition  la requte
		
		$req .= " AND "; // Ajoute le connecteur logique  la requte
		
		endswitch;
		
		}
		
		$req = substr($req, 0, -5); // Supprime le dernier " AND " de la requte
		
		$req .= $reqOrder; // Ajoute la clause ORDER BY  la requte
		
		}
		
		$response = $bdd->query($req); // Excute la requte de slection
		
		$data = $response->fetchAll(\PDO::FETCH_ASSOC); // Rcupre les donnes sous forme de tableau associatif
		
		return $data; // Retourne les donnes
		
		}
		
		  
		
		protected static function dbUpdate(string $table, array $data, string $idField = null)
		
		{
		
		$bdd = self::getDb(); // Rcupre la connexion  la base de donnes
		
		$req = "UPDATE " . $table . " SET "; // Construction de la requte de mise  jour
		
		$whereIdString = " WHERE id = :id"; // Condition WHERE pour l'identifiant
		
		// Set des key = :value
		
		foreach ($data as $key => $value) { // Parcours des donnes  mettre  jour
		
		if ($key !== 'id') { // Sauf si la cl est 'id'
		
		$req .= "" . $key . " = :" . $key . ", "; // Ajoute la cl et le placeholder  la requte
		
		}
		
		}
		
		$req = substr($req, 0, -2); // Supprime la virgule et l'espace  la fin de la requte
		
		$req .= $whereIdString; // Ajoute la condition WHERE  la requte
		
		$response = $bdd->prepare($req); // Prpare la requte
		
		$params = []; // Initialise le tableau de paramtres
		
		foreach ($data as $key => $value) { // Parcours des donnes  mettre  jour
		
		if ($key !== 'id') { // Sauf si la cl est 'id'
		
		$params[':' . $key] = $value; // Ajoute la cl et la valeur au tableau de paramtres
		
		}
		
		}
		
		$params[':id'] = $data['id']; // Ajoute l'identifiant au tableau de paramtres
		
		return $response->execute($params); // Excute la requte en liant les valeurs et retourne le rsultat
		
		}
	
	}

namespace Src\Models;
class Employe extends Model

	const TABLE_NAME = 'employes';

	$this->prenom = $prenom;

	$this->age = $age;
	
	}
	
	//
	
	  
	
	public function save()
	
	{
	
	$data = [
	
	"nom" => $this->getNom(),
	
	"prenom" => $this->getPrenom(),
	
	"age" => $this->getAge()
	
	];
	
	if ($this->id > 0)
	
	return $this->update();
	
	$nouvelId = Model::dbCreate(self::TABLE_NAME, $data);
	
	$this->setId($nouvelId);
	
	return $this;
	
	}
	
	  
	
	public function update()
	
	{
	
	if ($this->id > 0) {
	
	$data = [
	
	"id" => $this->getId(),
	
	"nom" => $this->getNom(),
	
	"prenom" => $this->getPrenom(),
	
	"age" => $this->getAge()
	
	];
	
	Model::dbUpdate(self::TABLE_NAME, $data, 'id');
	
	return $this;
	
	}
	
	return;
	
	}
	
	  
	
	public static function delete($id)
	
	{
	
	$data = [
	
	'id' => $id,
	
	];
	
	  
	
	Model::dbDelete(self::TABLE_NAME, $data);
	
	return;
	
	}
	
	  
	
	public static function findAll()
	
	{
	
	$data = Model::dbFind(self::TABLE_NAME);
	
	return $data;
	
	}
	
	  
	
	public static function find(array $request)

	$data = Model::dbFind(self::TABLE_NAME, $request);
	
	return $data;
	
	}
